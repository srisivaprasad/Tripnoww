#!/bin/bash
echo "Stopping Container.."
docker stop maincontainer
sleep 5

echo "Starting Container.."
docker start maincontainer
sleep 5

echo "Copying jar files from System to Container.."
docker cp /home/ubuntu/packages/ maincontainer:/opt/
sleep 5

echo "Executing Script from inside the Container.."
docker exec maincontainer /opt/start_services_zoopeeker_kafka.sh
